      function calcChannel(channel, label) {
        var data = {
          label: label,
          sales: readChannelInput(channel, "sales"),
          spend: readChannelInput(channel, "spend"),
          impressions: readChannelInput(channel, "impressions"),
          clicks: readChannelInput(channel, "clicks"),
          conversions: readChannelInput(channel, "conversions"),
          orders: readChannelInput(channel, "orders")
        };
        data.roas = data.spend > 0 ? Math.round((data.sales / data.spend) * 1000) / 10 : 0;
        data.ctr = data.impressions > 0 ? (data.clicks / data.impressions) * 100 : 0;
        data.cvr = data.clicks > 0 ? (data.conversions / data.clicks) * 100 : 0;
        data.cpa = data.conversions > 0 ? Math.round(data.spend / data.conversions) : 0;
        data.cpc = data.clicks > 0 ? Math.round(data.spend / data.clicks) : 0;
        return data;
      }

      function makeKeywordSeries(volume) {
        var base = Number(volume || 0);
        return [0.43, 0.74, 1, 0.91, 0.76, 0.62, 0.72, 0.79, 0.66, 0.53, 0.96, 0.58].map(function (ratio, index) {
          var wave = (((base / 100) + index * 17) % 23) / 100;
          return Math.max(800, Math.round(base * Math.min(1.08, ratio + wave)));
        });
      }

      function hashKeyword(keyword) {
        return String(keyword || "").split("").reduce(function (sum, char) {
          return sum + char.charCodeAt(0);
        }, 0);
      }

      function makeKeywordProfile(keyword, volume) {
        var seed = hashKeyword(keyword);
        var mobile = 58 + (seed % 34);
        var female = 38 + ((seed * 3) % 50);
        var age = [18, 30, 45, 62, 54, 28].map(function (value, index) {
          return Math.min(92, Math.max(12, value + ((seed + index * 11) % 24)));
        });
        var week = [70, 76, 82, 78, 72, 61, 66].map(function (value, index) {
          return Math.min(94, Math.max(35, value + ((seed + index * 7) % 18)));
        });
        return {
          series: makeKeywordSeries(volume),
          device: { mobile: mobile, pc: 100 - mobile },
          gender: { female: female, male: 100 - female },
          age: age,
          week: week
        };
      }

      function pickKeywordScale(maxValue) {
        var scales = [6000, 10000, 20000, 30000, 50000, 80000, 120000];
        for (var index = 0; index < scales.length; index += 1) {
          if (maxValue <= scales[index]) return scales[index];
        }
        return Math.ceil(maxValue / 50000) * 50000;
      }

      function setRatio(selector, label, value) {
        var node = root.querySelector(selector);
        if (!node) return;
        node.style.width = Math.max(6, Number(value || 0)) + "%";
        node.textContent = label + " " + Math.round(Number(value || 0)) + "%";
      }

      function setBars(selector, values) {
        root.querySelectorAll(selector).forEach(function (node) {
          var index = Number(node.getAttribute(selector.indexOf("age") > -1 ? "data-keyword-age-bar" : "data-keyword-week-bar"));
          node.style.height = Math.max(10, Number(values[index] || 0)) + "%";
        });
      }

      function renderKeywordChart(keyword, data) {
        var series = data.series || makeKeywordSeries(data.volume);
        var peakValue = Math.max.apply(null, series);
        var peakIndex = series.indexOf(peakValue);
        var chartMax = pickKeywordScale(peakValue);
        var width = 760;
        var height = 250;
        var top = 18;
        var bottom = 30;
        var points = series.map(function (value, index) {
          var x = series.length > 1 ? (index / (series.length - 1)) * width : 0;
          var y = height - bottom - ((value / chartMax) * (height - top - bottom));
          return x.toFixed(1) + "," + y.toFixed(1);
        });
        var peakX = series.length > 1 ? (peakIndex / (series.length - 1)) * width : 0;
        var peakY = height - bottom - ((peakValue / chartMax) * (height - top - bottom));
        var label = root.querySelector("[data-keyword-label]");
        var volume = root.querySelector("[data-keyword-volume]");
        var change = root.querySelector("[data-keyword-change]");
        var axis = root.querySelector("[data-keyword-axis]");
        var peak = root.querySelector("[data-keyword-peak]");
        var insight = root.querySelector("[data-keyword-insight]");
        var line = root.querySelector("[data-keyword-line]");
        var point = root.querySelector("[data-keyword-point]");
        var previous = series[series.length - 2] || data.volume;
        var changeRate = previous > 0 ? Math.round(((data.volume - previous) / previous) * 100) : 0;
        var profile = {
          device: data.device || { mobile: 85, pc: 15 },
          gender: data.gender || { female: 79, male: 21 },
          age: data.age || [18, 28, 58, 84, 70, 36],
          week: data.week || [82, 88, 86, 79, 74, 66, 76]
        };

        if (label) label.textContent = keyword;
        if (volume) volume.textContent = formatNumber(data.volume);
        if (change) change.textContent = (changeRate >= 0 ? "+" : "") + changeRate + "%";
        if (axis) axis.textContent = Math.round(chartMax / 1000) + "K";
        if (peak) peak.textContent = "2026.03 · " + formatNumber(peakValue);
        if (insight) insight.textContent = data.insight || "검색량과 클릭 비율을 기준으로 SEO/광고 후보를 분류";
        if (line) line.setAttribute("points", points.join(" "));
        if (point) {
          point.setAttribute("cx", peakX.toFixed(1));
          point.setAttribute("cy", peakY.toFixed(1));
        }
        setRatio("[data-keyword-device-mobile]", "모바일", profile.device.mobile);
        setRatio("[data-keyword-device-pc]", "PC", profile.device.pc);
        setRatio("[data-keyword-gender-female]", "여성", profile.gender.female);
        setRatio("[data-keyword-gender-male]", "남성", profile.gender.male);
        setBars("[data-keyword-age-bar]", profile.age);
        setBars("[data-keyword-week-bar]", profile.week);
      }

      function getKeywordApiUrl() {
        if (window.MI_NAVER_KEYWORD_API_URL) return window.MI_NAVER_KEYWORD_API_URL;
        var host = window.location && window.location.hostname;
        if (host && host !== "127.0.0.1" && host !== "localhost") return window.location.origin + "/api/naver-keyword";
        return "http://127.0.0.1:8787/api/naver-keyword";
      }

      function buildKeywordFallback(keyword) {
        var fallbackVolume = Math.max(1200, keyword.length * 2300);
        var profile = makeKeywordProfile(keyword, fallbackVolume);
        return keywordSamples[keyword] || {
          volume: fallbackVolume,
          comp: fallbackVolume > 25000 ? "높음" : "보통",
          action: "네이버 검색량 확인 후 SEO 후보로 분류",
          insight: "검색량 대비 경쟁도를 보고 네이버 SEO와 광고 키워드 우선순위 결정",
          series: profile.series,
          device: profile.device,
          gender: profile.gender,
          age: profile.age,
          week: profile.week
        };
      }

      function normalizeKeywordApiData(payload, fallbackData) {
        var chartData = payload && payload.chartData ? payload.chartData : {};
        var data = Object.assign({}, fallbackData, chartData);
        data.volume = Number(data.volume || fallbackData.volume || 0);
        data.comp = data.comp || fallbackData.comp || "확인 필요";
        data.action = data.action || fallbackData.action || "네이버 검색량 확인 후 SEO 후보로 분류";
        data.insight = data.insight || fallbackData.insight || "검색량과 클릭 비율을 기준으로 SEO/광고 후보를 분류";
        data.series = Array.isArray(data.series) && data.series.length ? data.series : fallbackData.series;
        data.device = data.device || fallbackData.device;
        data.gender = data.gender || fallbackData.gender;
        data.age = Array.isArray(data.age) && data.age.length ? data.age : fallbackData.age;
        data.week = Array.isArray(data.week) && data.week.length ? data.week : fallbackData.week;
        return data;
      }

      function keywordSourceLabel(payload, fallbackLabel) {
        var source = payload && payload.source ? payload.source : {};
        if (!source.searchVolume) return fallbackLabel;
        return source.trend === "NAVER DataLab API" ? "네이버 검색광고 API + 데이터랩 API" : "네이버 검색광고 API";
      }

      function renderKeywordResult(keyword, data, sourceLabel) {
        keywordResult.innerHTML =
          '<div class="mi-row"><span>검색 키워드</span><strong>' + escapeHtml(keyword) + '</strong></div>' +
          '<div class="mi-row"><span>월 검색량</span><strong>' + formatNumber(data.volume) + '</strong></div>' +
          '<div class="mi-row"><span>경쟁도</span><strong>' + escapeHtml(data.comp) + '</strong></div>' +
          '<div class="mi-row"><span>데이터 출처</span><strong>' + escapeHtml(sourceLabel) + '</strong></div>' +
          '<div class="mi-row"><span>추천 작업</span><strong>' + escapeHtml(data.action) + '</strong></div>';
      }

      async function fetchKeywordData(keyword) {
        if (!window.fetch) throw new Error("현재 브라우저에서 fetch를 사용할 수 없습니다.");
        var url = getKeywordApiUrl() + "?keyword=" + encodeURIComponent(keyword);
        var response = await fetch(url, { method: "GET", cache: "no-store" });
        var payload = await response.json().catch(function () {
          return null;
        });
        if (!response.ok || !payload || payload.ok === false) {
          throw new Error((payload && payload.message) || "네이버 키워드 API 응답이 올바르지 않습니다.");
        }
        return payload;
      }

      var keywordInput = root.querySelector("[data-admin-keyword-input]");
      var keywordButton = root.querySelector("[data-admin-keyword-search]");
      var keywordResult = root.querySelector("[data-admin-keyword-result]");

      renderKeywordChart("비타민 앰플", keywordSamples["비타민 앰플"]);
      if (keywordInput && keywordButton && keywordResult) {
        keywordButton.addEventListener("click", async function () {
          var keyword = keywordInput.value.trim() || "비타민 앰플";
          var fallbackData = buildKeywordFallback(keyword);
          keywordButton.disabled = true;
          setStatus(keywordStatus, "네이버 키워드 API 조회 중입니다.", "");
          try {
            var payload = await fetchKeywordData(keyword);
            var data = normalizeKeywordApiData(payload, fallbackData);
            renderKeywordChart(keyword, data);
            renderKeywordResult(keyword, data, keywordSourceLabel(payload, "네이버 API"));
            setStatus(keywordStatus, "조회 완료: 네이버 검색 기반 실데이터를 반영했습니다.", "is-ok");
          } catch (error) {
            renderKeywordChart(keyword, fallbackData);
            renderKeywordResult(keyword, fallbackData, "데모 계산값");
            setStatus(keywordStatus, "네이버 API 미연결: " + error.message + " 데모 계산값으로 표시합니다.", "is-warn");
          } finally {
            keywordButton.disabled = false;
          }
        });
      }

      function csvValue(value) {
        return '"' + String(value || "").replace(/"/g, '""') + '"';
      }

      function buildReportCsv(type, state) {
        var rows = [
          ["보고서 유형", type],
          ["광고주", state.client],
          ["대행사 코드", state.code],
          ["공개 매출", state.sales],
          ["ROAS", state.roas],
          ["광고비", state.adSpend],
          ["구매수", state.orders],
          ["목표 달성률", state.achievement],
          ["상태", state.status],
          ["다음 일정", state.nextSchedule],
          ["공개 코멘트", state.comment],
          [],
          ["채널", "요약", "매출", "ROAS"]
        ];
        (state.channels || []).forEach(function (item) {
          rows.push([item.name, item.summary, item.sales, item.roas]);
        });
        rows.push([]);
        rows.push(["날짜", "일정", "상세", "상태"]);
        (state.schedules || []).forEach(function (item) {
          rows.push([item.date, item.title, item.detail, item.status]);
        });
        return rows.map(function (row) {
          return row.map(csvValue).join(",");
        }).join("\n");
      }

      root.querySelectorAll("[data-admin-download]").forEach(function (button) {
        button.addEventListener("click", function () {
          var original = button.innerHTML;
          var type = button.getAttribute("data-report-type") || "report";
          var state = readState();
          var csv = buildReportCsv(type, state);
          var blob = new Blob(["\ufeff" + csv], { type: "text/csv;charset=utf-8" });
          var url = URL.createObjectURL(blob);
          var link = document.createElement("a");
          link.href = url;
          link.download = "moment-insight-" + type + "-" + state.client + ".csv";
          document.body.appendChild(link);
          link.click();
          link.remove();
          setTimeout(function () {
            URL.revokeObjectURL(url);
          }, 1000);
          button.innerHTML = "<span><b>CSV 생성 완료</b><small>관리자가 확인 후 광고주에게 전달</small></span><i>✓</i>";
          setTimeout(function () {
            button.innerHTML = original;
          }, 1700);
        });
      });
