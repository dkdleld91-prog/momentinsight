      async function fetchDemoState() {
        if (!window.fetch) throw new Error("현재 브라우저에서 fetch를 사용할 수 없습니다.");
        var publicStateUrl = window.MI_PUBLIC_STATE_API_URL || "";
        if (!publicStateUrl) throw new Error("공개 데이터 API가 설정되지 않았습니다.");
        var response = await fetch(publicStateUrl, { method: "GET", cache: "no-store" });
        var payload = await response.json().catch(function () {
          return null;
        });
        if (!response.ok || !payload || payload.ok === false || !payload.publicState) {
          throw new Error((payload && payload.message) || "브랜드 데이터 응답이 올바르지 않습니다.");
        }
        return enrichState(payload.publicState);
      }

      function escapeHtml(value) {
        return String(value || "")
          .replace(/&/g, "&amp;")
          .replace(/</g, "&lt;")
          .replace(/>/g, "&gt;")

      function parseKeywordInput(value) {
        var seen = {};
        return String(value || "")
          .split(/[\n,，]+/)
          .map(function (keyword) { return keyword.trim(); })
          .filter(function (keyword) {
            if (!keyword || seen[keyword]) return false;
            seen[keyword] = true;
            return true;
          })
          .slice(0, 1);
      }

      function pickKeywordScale(maxValue) {
        var scales = [6000, 10000, 20000, 30000, 50000, 80000, 120000, 200000, 300000, 500000];
        for (var index = 0; index < scales.length; index += 1) {
          if (maxValue <= scales[index]) return scales[index];
        }
        return Math.ceil(maxValue / 50000) * 50000;
      }

      function formatShortNumber(value) {
        var number = Number(value || 0);
        if (number >= 1000000) return Math.round(number / 1000000) + "M";
        if (number >= 1000) return Math.round(number / 1000) + "K";
        return formatNumber(number);
      }

      function formatAxisTick(value) {
        var number = Number(value || 0);
        if (number >= 1000000) {
          var million = number / 1000000;
          return (Number.isInteger(million) ? String(million) : million.toFixed(1)) + "M";
        }
        if (number >= 1000) {
          var thousand = number / 1000;
          return (Number.isInteger(thousand) ? String(thousand) : thousand.toFixed(1)) + "K";
        }
        return formatNumber(number);
      }

      function makeKeywordMonthLabels(count) {
        var now = new Date();
        var labels = [];
        for (var index = count - 1; index >= 0; index -= 1) {
          var date = new Date(now.getFullYear(), now.getMonth() - index, 1);
          var label = String(date.getMonth() + 1).padStart(2, "0") + "월";
          labels.push(index === 0 ? label + "(예상)" : label);
        }
        return labels;
      }

      function renderKeywordMonths(labels) {
        var node = root.querySelector("[data-keyword-months]");
        if (!node) return;
        var maxLabels = 12;
        var visibleLabels = labels.map(function (label, index) {
          return { label: label, index: index };
        });
        if (labels.length > maxLabels) {
          var step = Math.ceil(labels.length / maxLabels);
          visibleLabels = visibleLabels.filter(function (item) {
            return item.index % step === 0 || item.index === labels.length - 1;
          });
        }
        node.innerHTML = visibleLabels.map(function (item) {
          var left = labels.length > 1 ? ((item.index + 0.5) / labels.length) * 100 : 50;
          return '<span style="left:' + left.toFixed(3) + '%;">' + escapeHtml(item.label) + '</span>';
        }).join("");
      }

      function normalizePercentValues(values) {
        var source = values.map(function (value) {
          return Math.max(0, Number(value || 0));
        });
        var total = source.reduce(function (sum, value) {
          return sum + value;
        }, 0);
        if (!total) {
          return source.map(function () { return 0; });
        }
        return source.map(function (value) {
          return Math.round((value / total) * 1000) / 10;
        });
      }

          return;
        }
        var axisMax = ratioAxisMax(values, minimumAxis);
        var step = axisMax <= 20 ? 5 : 10;
        var ticks = [];
        for (var tick = axisMax; tick >= 0; tick -= step) ticks.push(tick);
        if (ticks[ticks.length - 1] !== 0) ticks.push(0);
        var peak = Math.max.apply(null, values);
        chart.style.setProperty("--ratio-cols", labels.length);
        chart.innerHTML = '<div class="mi-ratio-axis">' + ticks.map(function (value) {
          var top = axisMax ? ((axisMax - value) / axisMax) * 100 : 100;
          return '<span style="top:' + top.toFixed(2) + '%;">' + value + '%</span>';
        }).join("") + '</div><div class="mi-ratio-bars">' + labels.map(function (label, index) {
          var value = values[index] || 0;
          var height = axisMax ? Math.max(3, (value / axisMax) * 100) : 3;
          var isPeak = value === peak ? " is-peak" : "";
          return '<span class="mi-ratio-column' + isPeak + '" style="--ratio-height:' + height.toFixed(2) + '%;">' +

            '<i style="height:' + height.toFixed(2) + '%;"></i><b>' + escapeHtml(label) + '</b></span>';
        }).join("") + '</div>';
        chart.querySelectorAll(".mi-ratio-column").forEach(function (column) {
          column.addEventListener("mouseenter", function () { column.classList.add("is-hover"); });

      function keywordGrade(score) {
        if (!Number.isFinite(score) || score <= 0) return { label: "확인 필요", hint: "상품/검색 확인", className: "warn" };
        if (score <= 1) return { label: "낮음", hint: "경쟁강도", className: "" };
        if (score <= 3) return { label: "보통", hint: "경쟁강도", className: "" };
        if (score <= 6) return { label: "높음", hint: "경쟁강도", className: "" };
        return { label: "매우 높음", hint: "경쟁강도", className: "" };
      }

      function gradeFromCompetition(comp, fallbackScore) {
        var label = String(comp || "");

      }

      function renderKeywordChart(itemsOrKeyword, singleData) {
        var items = Array.isArray(itemsOrKeyword) ? itemsOrKeyword : [{ keyword: itemsOrKeyword, data: singleData }];
        items = items.filter(function (item) { return item && item.keyword && item.data; }).slice(0, 1);
        if (!items.length) {
          clearKeywordDataView();
          return;
        }

        keywordDatasets = items;
        var requestedCount = keywordRangeMonths[keywordRange] || 12;
        var seriesList = items.map(function (item) {
          return rangeSeries(item.keyword, item.data);
        });
        var flatSeries = [].concat.apply([], seriesList);
        var hasTrendSeries = flatSeries.some(function (value) {
          return Number(value || 0) > 0;
        });
        var effectiveCount = hasTrendSeries ? Math.max.apply(null, seriesList.map(function (series) {
          return series.length;
        }).concat([1])) : requestedCount;
        var labelItem = items.find(function (item) {
          return item && item.data && Array.isArray(item.data.seriesLabels) && item.data.seriesLabels.length;
        }) || items[0];
        var labels = rangeLabels(labelItem.data, effectiveCount);
        if (labels.length !== effectiveCount) labels = makeKeywordMonthLabels(effectiveCount);
        var totalVolume = items.reduce(function (sum, item) { return sum + Number(item.data.volume || 0); }, 0);
        var quotaLimited = items.some(function (item) {
          return String(item.data && item.data.userMessage || "").indexOf("한도") > -1;
        });
        var peakValue = hasTrendSeries ? Math.max.apply(null, flatSeries) : 0;
        var chartMax = hasTrendSeries ? pickKeywordScale(peakValue) : 0;
        var width = 760;
        var height = 250;
        var top = 18;
        var bottom = 30;
        var currentTrendTotal = seriesList.reduce(function (sum, series) {
          return sum + Number(series[series.length - 1] || 0);
        }, 0);
        var previousTrendTotal = seriesList.reduce(function (sum, series) {
          return sum + Number(series[series.length - 2] || series[series.length - 1] || 0);
        }, 0);
        var peakInfo = { keyword: items[0].keyword, label: labels[0], value: peakValue };
        var label = root.querySelector("[data-keyword-label]");
        var volumeTitle = root.querySelector("[data-keyword-volume-title]");
        var volume = root.querySelector("[data-keyword-volume]");
        var change = root.querySelector("[data-keyword-change]");
        var axis = root.querySelector("[data-keyword-axis]");
        var peak = root.querySelector("[data-keyword-peak]");
        var insight = root.querySelector("[data-keyword-insight]");
        var legend = root.querySelector("[data-keyword-legend]");
        var linesGroup = root.querySelector("[data-keyword-lines]");
        var hoverGroup = root.querySelector("[data-keyword-hover]");
        var pointsGroup = root.querySelector("[data-keyword-points]");
        var empty = root.querySelector("[data-keyword-chart-empty]");
        var changeRate = previousTrendTotal > 0 ? Math.round(((currentTrendTotal - previousTrendTotal) / previousTrendTotal) * 100) : 0;
        var profile = {
          device: firstKeywordProfile(items, "device", null),
          gender: firstKeywordProfile(items, "gender", null),
          month: firstKeywordProfile(items, "month", []),
          age: firstKeywordProfile(items, "age", []),
          week: firstKeywordProfile(items, "week", [])
        };

        if (label) label.textContent = items.length > 1 ? items[0].keyword + " 외 " + (items.length - 1) + "개" : items[0].keyword;
        if (volumeTitle) volumeTitle.textContent = items.length > 1 ? "월 검색수" : "월 검색수";
        if (volume) volume.textContent = totalVolume ? formatNumber(totalVolume) : "확인 필요";
        if (change) change.textContent = hasTrendSeries ? (changeRate >= 0 ? "+" : "") + changeRate + "%" : "조회 후 표시";
        if (axis) axis.textContent = hasTrendSeries ? formatShortNumber(chartMax) : "";
        if (legend) {
          legend.innerHTML = items.map(function (item, index) {
          return '<span><i class="mi-dot" style="--keyword-color:' + keywordColors[index % keywordColors.length] + '"></i>' + escapeHtml(item.keyword) + ' ' + escapeHtml(formatSearchVolume(item.data)) + '</span>';
          }).join("");
        }
        if (linesGroup) linesGroup.innerHTML = "";
        if (hoverGroup) hoverGroup.innerHTML = "";
        if (pointsGroup) pointsGroup.innerHTML = "";
        if (empty) {
          empty.hidden = hasTrendSeries;
          empty.textContent = quotaLimited ? "월별 검색량 추이는 DataLab 한도 복구 후 표시됩니다." : totalVolume ? "월별 검색량 추이는 연결 후 표시됩니다." : "검색량 조회 후 차트를 표시합니다.";
        }

        var seriesPointLists = hasTrendSeries ? seriesList.map(function (series, seriesIndex) {
          return series.map(function (value, index) {
            var x = series.length > 1 ? ((index + 0.5) / series.length) * width : width / 2;
            var y = height - bottom - ((value / chartMax) * (height - top - bottom));
            if (value >= peakInfo.value) peakInfo = { keyword: items[seriesIndex].keyword, label: labels[index] || "", value: value };
            return { x: x, y: y, value: value, label: labels[index] || "" };
          });
        }) : [];

        seriesPointLists.forEach(function (pointPairs, seriesIndex) {
          var color = keywordColors[seriesIndex % keywordColors.length];
          if (linesGroup) {
            var line = document.createElementNS("http://www.w3.org/2000/svg", "polyline");

            }).join(" "));

            line.setAttribute("stroke-width", "2.4");
            line.setAttribute("stroke-linecap", "round");
            line.setAttribute("stroke-linejoin", "round");
            line.setAttribute("opacity", seriesIndex === 0 ? "1" : "0.9");
            linesGroup.appendChild(line);
          }
        });

        if (pointsGroup && hasTrendSeries && seriesPointLists[0]) {
          seriesPointLists[0].forEach(function (basePoint, pointIndex) {
            var pointCount = seriesPointLists[0].length || 1;
            var startX = (pointIndex / pointCount) * width;
            var endX = ((pointIndex + 1) / pointCount) * width;
            var hoverPoints = seriesPointLists.map(function (points, seriesIndex) {
              var point = points[pointIndex] || basePoint;
              return {
                x: basePoint.x,
                y: point.y,
                value: point.value,
                label: basePoint.label,
                keyword: items[seriesIndex].keyword,
                color: keywordColors[seriesIndex % keywordColors.length]
              };
            });
            var tooltipText = "<b>" + escapeHtml(basePoint.label) + " 검색량</b>" + hoverPoints.map(function (point) {
              return '<span style="display:block;"><i class="mi-dot" style="--keyword-color:' + point.color + '"></i>' + escapeHtml(point.keyword) + " " + formatMonthlySearchPoint(point.value) + "</span>";
            }).join("");
            var hit = document.createElementNS("http://www.w3.org/2000/svg", "rect");
            hit.setAttribute("class", "mi-chart-hit");
            hit.setAttribute("x", startX.toFixed(1));
            hit.setAttribute("y", "0");
            hit.setAttribute("width", Math.max(12, endX - startX).toFixed(1));
            hit.setAttribute("height", height.toString());
            hit.setAttribute("fill", "transparent");
            hit.addEventListener("mouseenter", function (event) {
              showKeywordHover(event, hoverPoints, tooltipText);
            });
            hit.addEventListener("mousemove", function (event) {
              showKeywordHover(event, hoverPoints, tooltipText);
            });
            hit.addEventListener("mouseleave", hideKeywordTooltip);
            pointsGroup.appendChild(hit);
          });
        }

        if (peak) peak.textContent = "";
        if (insight) {
          insight.textContent = items[0].data.insight || "검색량과 검색 비율을 기준으로 SEO/소재 후보를 분류";
        }
        renderKeywordMarket(items);
        renderKeywordMonths(labels);
        renderKeywordYAxis(hasTrendSeries ? chartMax : 0);
        renderProfileSplit("device", "모바일", "PC", profile.device && profile.device.mobile, profile.device && profile.device.pc, "조회 후 표시");
        renderProfileSplit("gender", "여성", "남성", profile.gender && profile.gender.female, profile.gender && profile.gender.male, "조회 후 표시");
        var monthSeries = hasRatioValues(profile.month) ? profile.month : [];
        var monthLabels = alignedMonthLabels(items[0].data, monthSeries, []);
        var ageValues = Array.isArray(profile.age) ? profile.age.slice(0, 5).map(function (value) {
          return Number(value || 0);
        }) : [];
        if (Array.isArray(profile.age) && profile.age.length > 5) {
          ageValues[4] = profile.age.slice(4).reduce(function (sum, value) {
            return sum + Number(value || 0);
          }, 0);
        }
        if (hasRatioValues(monthSeries)) {
          renderRatioChart("month", monthLabels, monthSeries, 30);
        } else {
          renderRatioPlaceholder("month", "조회 후 표시");
        }
        if (hasRatioValues(profile.week)) {
          renderRatioChart("week", ["월", "화", "수", "목", "금", "토", "일"], profile.week, 20);
        } else {
          renderRatioPlaceholder("week", "조회 후 표시");

        }
      }

      function formatRankCheckedAt(iso) {
        if (!iso) return "-";
        try {
          return new Date(iso).toLocaleString("ko-KR", { month: "2-digit", day: "2-digit", hour: "2-digit", minute: "2-digit" });
        } catch (error) {
          return "-";
        }
      }

      function formatRankShortAt(iso) {
        if (!iso) return "-";
        try {
          return new Date(iso).toLocaleString("ko-KR", { month: "2-digit", day: "2-digit", hour: "2-digit", minute: "2-digit" });
        } catch (error) {
          return "-";
        }
      }

      function formatRankRemain(iso) {
        if (!iso) return "-";
        var diff = new Date(iso).getTime() - Date.now();
        if (!Number.isFinite(diff)) return "-";
        if (diff <= 0) return "대기";
        var minutes = Math.ceil(diff / 60000);
        if (minutes < 60) return minutes + "분";
        var hours = Math.ceil(minutes / 60);
        if (hours < 24) return hours + "시간";
        return Math.ceil(hours / 24) + "일";
      }

      function rankText(value) {

      function rankTrackerStatus(tracker) {
        if (!tracker) return "-";
        if (tracker.status === "paused") return "중지";
        if (tracker.status === "completed") return "종료";
        if (tracker.status === "failed") return "오류";

      }

      function rankDateKey(date) {
        return date.getFullYear() + "-" + String(date.getMonth() + 1).padStart(2, "0") + "-" + String(date.getDate()).padStart(2, "0");
      }

      function rankDayLabel(date) {

      }

      function groupRankSnapshotsByDay(snapshots) {
        var grouped = {};
        (snapshots || []).forEach(function (snapshot) {
          if (!snapshot || !snapshot.checkedAt) return;
          var date = new Date(snapshot.checkedAt);
          if (Number.isNaN(date.getTime())) return;
          var key = rankDateKey(date);
          var slot = date.getHours() < 12 ? "am" : "pm";
          if (!grouped[key]) grouped[key] = { date: date, am: null, pm: null };
          if (!grouped[key][slot] || new Date(grouped[key][slot].checkedAt).getTime() < date.getTime()) {
            grouped[key][slot] = snapshot;
          }
        });
        return grouped;
      }

      function renderRankSlot(label, snapshot) {

        return '<span' + (!snapshot || !snapshot.rank ? ' class="is-empty"' : "") + '><small>' + label + '</small><b>' + escapeHtml(rank) + '</b></span>';
      }

      function renderTrackerDailyBoard(tracker) {
        var grouped = groupRankSnapshotsByDay(tracker.snapshots || []);
        var keys = Object.keys(grouped).sort(function (a, b) {
          return new Date(grouped[b].date).getTime() - new Date(grouped[a].date).getTime();
        }).slice(0, 30);
        if (!keys.length) {
          return '<div class="mi-rank-state">첫 순위 확인 후 이 상품만의 일별 기록판이 생성됩니다.</div>';
        }

      function initRankTracking(linkedKeywordInput) {
        var card = root.querySelector("[data-rank-card]");
        if (!card) return;
        var rankKeyword = card.querySelector("[data-rank-keyword]");
        var rankUrl = card.querySelector("[data-rank-url]");
        var rankProductId = card.querySelector("[data-rank-product-id]");
        var rankMax = card.querySelector("[data-rank-max]");
        var rankRun = card.querySelector("[data-rank-run]");
        var rankStatus = card.querySelector("[data-rank-status]");
        var rankResult = card.querySelector("[data-rank-result]");
        var rankHistory = card.querySelector("[data-rank-history]");
        if (!rankKeyword || !rankRun) return;
        var rankTrackers = [];

        async function requestRankTrackers(method, body) {
          var options = { method: method, cache: "no-store" };
          if (body) {
            options.headers = { "content-type": "application/json" };
            options.body = JSON.stringify(Object.assign({ agencyCode: rankAgencyCode() }, body));
          }
          var url = getRankTrackerApiUrl();
          if (method === "GET") {
            url += "?" + new URLSearchParams({ agencyCode: rankAgencyCode(), limit: "20" }).toString();
          }
          var response = await fetch(url, options);
          var payload = await response.json().catch(function () { return null; });
          if (!payload) payload = { ok: false, message: "순위 추적 응답을 확인할 수 없습니다." };
          if (!response.ok && payload.ok !== false) payload.ok = false;
          return payload;
        }

        async function loadRankTrackers(silent) {
          try {
            var payload = await requestRankTrackers("GET");
            rankTrackers = payload.trackers || [];
            renderRankResult(rankResult, { ok: payload.ok, trackers: rankTrackers, message: payload.message });
            renderRankHistory(rankHistory, rankTrackers);
            if (!silent) {
              setStatus(rankStatus, rankTrackers.length ? "등록된 상품 순위 추적을 불러왔습니다." : "상품 URL 또는 상품ID를 입력하면 30일 추적을 시작합니다.", rankTrackers.length ? "is-ok" : "");
            }
          } catch (error) {
            renderRankResult(rankResult, { ok: false, message: "순위 추적 서버 연결을 확인해주세요." });
            renderRankHistory(rankHistory, []);
            if (!silent) setStatus(rankStatus, "순위 추적 서버 연결을 확인해주세요.", "is-warn");
          }
        }

        function syncKeywordFromMain() {
          if (!linkedKeywordInput || !rankKeyword) return;
          var next = parseKeywordInput(linkedKeywordInput.value)[0] || "";
          if (next && (!rankKeyword.value || rankKeyword.getAttribute("data-synced-keyword") === "true")) {
            rankKeyword.value = next;
            rankKeyword.setAttribute("data-synced-keyword", "true");
          }
        }

        rankKeyword.setAttribute("data-synced-keyword", "true");
        syncKeywordFromMain();
        loadRankTrackers(false);

        if (linkedKeywordInput) {
          linkedKeywordInput.addEventListener("input", syncKeywordFromMain);
        }
        rankKeyword.addEventListener("input", function () {
          rankKeyword.setAttribute("data-synced-keyword", "false");
        });

        rankRun.addEventListener("click", async function () {
          var keyword = (rankKeyword.value || (linkedKeywordInput && linkedKeywordInput.value) || "").trim();
          var targetUrl = rankUrl ? rankUrl.value.trim() : "";
          var productId = rankProductId ? rankProductId.value.trim() : "";
          if (!keyword) {
            setStatus(rankStatus, "키워드를 입력해주세요.", "is-warn");
            return;
          }
          if (!targetUrl && !productId) {
            setStatus(rankStatus, "네이버 상품 URL 또는 상품ID를 입력해주세요.", "is-warn");
            return;
          }

          rankRun.disabled = true;
          setStatus(rankStatus, "30일 순위 추적을 등록하고 첫 순위를 확인 중입니다.", "");
          try {
            var payload = await requestRankTrackers("POST", {
              action: "create",
              keyword: keyword,
              targetUrl: targetUrl,
              productId: productId,
              maxRank: rankMax ? rankMax.value : "300"
            });
            if (payload.tracker) {
              rankTrackers = [payload.tracker].concat(rankTrackers.filter(function (item) { return item.id !== payload.tracker.id; }));
            }
            renderRankResult(rankResult, { ok: payload.ok, trackers: rankTrackers, message: payload.message });
            renderRankHistory(rankHistory, rankTrackers);
            setStatus(rankStatus, payload.message || "순위 추적 등록 완료", payload.ok ? "is-ok" : "is-warn");
            await loadRankTrackers(true);
          } catch (error) {
            renderRankResult(rankResult, { ok: false, message: "순위 추적 서버 연결을 확인해주세요." });
            setStatus(rankStatus, "순위 추적 서버 연결을 확인해주세요.", "is-warn");
          } finally {
            rankRun.disabled = false;
          }
        });

        card.addEventListener("click", async function (event) {
          var refreshButton = event.target.closest("[data-rank-refresh]");
          if (!refreshButton) return;
          refreshButton.disabled = true;
          setStatus(rankStatus, "선택한 상품 순위를 수동 갱신 중입니다.", "");
          try {
            var payload = await requestRankTrackers("POST", {
              action: "check",
              trackerId: refreshButton.getAttribute("data-rank-refresh")
            });
            setStatus(rankStatus, payload.message || "수동 갱신 완료", payload.ok ? "is-ok" : "is-warn");
            await loadRankTrackers(true);
          } catch (error) {
            setStatus(rankStatus, "수동 갱신 연결을 확인해주세요.", "is-warn");
          } finally {
            refreshButton.disabled = false;
          }
        });
      }

      function normalizeKeywordApiData(payload, fallbackData) {
        var chartData = payload && payload.chartData ? payload.chartData : {};
        var data = Object.assign({}, fallbackData, chartData);
        data.volume = Number(data.volume ?? fallbackData.volume ?? 0);
        data.volumeLabel = chartData.volumeLabel || fallbackData.volumeLabel || "";
        data.volumeStatus = chartData.volumeStatus || (data.volume ? "확인됨" : "확인 필요");
        data.trendStatus = chartData.trendStatus || ((payload && payload.source && (payload.source.trend === "naver_datalab_search" || payload.source.trend === "naver_datalab_relative_ratio")) ? "확인됨" : "조회 후 표시");
        data.comp = data.comp || fallbackData.comp || "확인 필요";
        data.action = data.action || fallbackData.action || "네이버 검색량 확인 후 SEO 후보로 분류";
        data.insight = data.insight || fallbackData.insight || "검색량과 검색 비율을 기준으로 SEO/소재 후보를 분류";
        data.series = Array.isArray(data.series) && data.series.length ? data.series : fallbackData.series;
        data.seriesLabels = Array.isArray(data.seriesLabels) && data.seriesLabels.length ? data.seriesLabels : fallbackData.seriesLabels;
        data.trendUnit = data.trendUnit || fallbackData.trendUnit || "";
        data.month = Array.isArray(data.month) && data.month.length ? data.month : fallbackData.month;
        data.monthLabels = Array.isArray(data.monthLabels) && data.monthLabels.length ? data.monthLabels : fallbackData.monthLabels;
        data.device = data.device || fallbackData.device;
        data.gender = data.gender || fallbackData.gender;
        data.age = Array.isArray(data.age) && data.age.length ? data.age : fallbackData.age;
        data.demographicStatus = data.demographicStatus || fallbackData.demographicStatus || "";
        data.week = Array.isArray(data.week) && data.week.length ? data.week : fallbackData.week;
        data.sourceStatus = payload && payload.sourceStatus ? payload.sourceStatus : {};
        data.userMessage = payload && payload.userMessage ? payload.userMessage : "";
        return data;
      }

      function keywordSourceLabel(payload, fallbackLabel) {
        var sourceStatus = payload && payload.sourceStatus ? payload.sourceStatus : {};
        if (sourceStatus.searchVolume && sourceStatus.searchVolume.label) {
          var labels = [sourceStatus.searchVolume.label];
          if (sourceStatus.ratios && sourceStatus.ratios.status !== "ok") labels.push(sourceStatus.ratios.label);
          if (sourceStatus.shopping && sourceStatus.shopping.status !== "ok") labels.push(sourceStatus.shopping.label);
          return labels.filter(Boolean).join(" · ");
        }
        var source = payload && payload.source ? payload.source : {};
        if (!source.searchVolume) return fallbackLabel;
        if (payload && payload.chartData && payload.chartData.volumeStatus === "확인됨") return "월 검색량 확인됨";
        if (payload && payload.chartData && payload.chartData.volumeStatus === "범위 확인") return "검색량 범위 확인";
        return "확인 필요";
      }

      async function fetchKeywordData(keyword, profileMode) {
        if (!window.fetch) throw new Error("현재 브라우저에서 fetch를 사용할 수 없습니다.");
        var url = getKeywordApiUrl() + "?keyword=" + encodeURIComponent(keyword) + "&profile=" + encodeURIComponent(profileMode || "full");
        var response = await fetch(url, { method: "GET", cache: "no-store" });
        var payload = await response.json().catch(function () {
          return null;
        });
        if (!response.ok || !payload || payload.ok === false) {
          throw new Error((payload && payload.message) || "키워드 데이터 응답이 올바르지 않습니다.");
        }
        return payload;
      }

      async function resolveKeywordItem(keyword, index) {

          data: normalizeKeywordApiData(payload, fallbackData),
          sourceLabel: keywordSourceLabel(payload, "월간 검색 데이터"),
          ok: true
        };
      }

      function unlockWithCode(code, statusNode) {
        var state = readState();
        var matched = normalizeCode(code) === normalizeCode(state.code);
        if (!matched) {
          setStatus(statusNode, "코드가 일치하지 않습니다. 관리자에게 받은 코드를 확인하세요.", "is-warn");
          return false;
        }
        try {
          window.localStorage.setItem(selectedCodeKey, state.code);
        } catch (error) {}
        applyState(state);
        root.classList.remove("is-locked");
        root.classList.add("is-authed");
        setStatus(statusNode, "연결 완료: " + state.client + " 브랜드 데이터가 반영되었습니다.", "is-ok");
        return true;
      }

      function restoreAuth(state) {
        try {
          var selectedCode = window.localStorage.getItem(selectedCodeKey);
          if (normalizeCode(selectedCode) !== normalizeCode(state.code)) return;
          root.classList.remove("is-locked");
          root.classList.add("is-authed");
        } catch (error) {}
      }

      function setScreen(name, shouldPushHash) {
        var target = name || "dashboard";
        var found = false;

        views.forEach(function (view) {
          var active = view.getAttribute("data-mi-view") === target;
          view.classList.toggle("is-active", active);
          if (active) found = true;
        });

        if (!found) {
          target = "dashboard";
          views.forEach(function (view) {
            view.classList.toggle("is-active", view.getAttribute("data-mi-view") === target);
          });
        }

        links.forEach(function (link) {
          link.classList.toggle("is-active", link.getAttribute("data-mi-screen") === target);
        });

        if (shouldPushHash) {
          var nextHash = "#mi-" + target;
          if (window.location.hash !== nextHash) window.location.hash = nextHash;
        }
      }

      links.forEach(function (link) {
        link.addEventListener("click", function (event) {
          event.preventDefault();
          setScreen(link.getAttribute("data-mi-screen"), true);
        });
      });

        var next = (window.location.hash || "").replace("#mi-", "");

      }

      var connectCode = root.querySelector("[data-mi-connect-code]");
      var connectButton = root.querySelector("[data-mi-connect-button]");
      var connectStatus = root.querySelector("[data-mi-connect-status]");
      if (connectButton && connectCode) {
        connectButton.addEventListener("click", async function () {
          connectButton.disabled = true;
          await syncDemoState(connectStatus);
          unlockWithCode(connectCode.value, connectStatus);
          connectButton.disabled = false;
        });
      }

      var keywordInput = root.querySelector("[data-mi-keyword-input]");
      var keywordButton = root.querySelector("[data-mi-keyword-search]");
      var keywordResult = root.querySelector("[data-mi-keyword-result]");
      var keywordStatus = root.querySelector("[data-mi-keyword-status]");
      var keywordRangeButtons = root.querySelectorAll("[data-keyword-range]");
      keywordDatasets = [];
      clearKeywordDataView("키워드를 입력하고 조회를 누르면 실제 API 데이터만 표시합니다.");
      renderKeywordResult(keywordDatasets);
      keywordRangeButtons.forEach(function (button) {
        button.addEventListener("click", function () {
          keywordRange = button.getAttribute("data-keyword-range") || "year";
          keywordRangeButtons.forEach(function (node) {
            node.classList.toggle("is-active", node === button);
          });
          renderKeywordChart(keywordDatasets);
        });
      });
      async function runKeywordLookup() {
        if (!keywordInput || !keywordButton || !keywordResult) return;
          var keywords = parseKeywordInput(keywordInput.value);
          if (!keywords.length) {
            keywordDatasets = [];
            clearKeywordDataView("키워드를 입력하고 조회를 눌러주세요.");
            renderKeywordResult(keywordDatasets);
            setStatus(keywordStatus, "키워드를 입력해주세요.", "is-warn");
            return;
          }
          keywordButton.disabled = true;
          setStatus(keywordStatus, "키워드 데이터를 조회 중입니다. 1개 키워드 기준으로 계산합니다.", "");
          try {
            var items = await Promise.all(keywords.map(resolveKeywordItem));
            keywordDatasets = items;
            renderKeywordChart(items);
            renderKeywordResult(items);
            var failed = items.filter(function (item) { return !item.ok; });
            var pendingTrend = items.filter(function (item) { return item.data.trendStatus !== "확인됨"; });
            var missingVolume = items.filter(function (item) { return !hasSearchVolume(item.data); });
            var primaryProfile = items[0] && items[0].data ? items[0].data : {};
            var primaryMessage = primaryProfile.userMessage || "";
            var hasDeviceProfile = primaryProfile.device && (Number(primaryProfile.device.mobile || 0) > 0 || Number(primaryProfile.device.pc || 0) > 0);
            var hasGenderProfile = primaryProfile.gender && (Number(primaryProfile.gender.female || 0) > 0 || Number(primaryProfile.gender.male || 0) > 0);
            var missingProfile = !hasRatioValues(primaryProfile.month || []) || !hasRatioValues(primaryProfile.week || []) || !hasRatioValues(primaryProfile.age || []) || !hasDeviceProfile || !hasGenderProfile;
            if (failed.length) {
              setStatus(keywordStatus, "해당 키워드는 검색량 확인이 필요합니다. 확인된 수치만 표시합니다.", "is-warn");
            } else if (missingVolume.length) {
              setStatus(keywordStatus, "조회 완료: 해당 키워드는 정확한 월 검색량이 확인되지 않았습니다.", "is-warn");
            } else if (primaryMessage.indexOf("한도") > -1) {
              setStatus(keywordStatus, primaryMessage, "is-warn");
            } else if (pendingTrend.length) {
              setStatus(keywordStatus, "조회 완료: 월 검색량은 확인됐고, 비율 데이터는 연결 후 표시됩니다.", "is-ok");
            } else if (missingProfile) {
              setStatus(keywordStatus, "조회 완료: 일부 검색 비율 데이터는 연결 후 표시됩니다.", "is-warn");
            } else {
              setStatus(keywordStatus, "조회 완료: 월 검색량과 검색 비율을 1개 키워드 기준으로 표시합니다.", "is-ok");
            }
          } catch (error) {
            keywordDatasets = [];
            clearKeywordDataView("키워드 데이터 연결을 확인해주세요.");
            renderKeywordResult(keywordDatasets);
            setStatus(keywordStatus, error.message || "키워드 데이터 연결을 확인해주세요.", "is-warn");
          } finally {
            keywordButton.disabled = false;
          }
      }

      if (keywordInput && keywordButton && keywordResult) {
        keywordButton.addEventListener("click", function () {
          runKeywordLookup();
        });
      }
      initRankTracking(keywordInput);

      var state = readState();
      applyState(state);
      if (loginCode) loginCode.value = state.code;
      if (connectCode) connectCode.value = state.code;
      restoreAuth(state);

      var initial = (window.location.hash || "").replace("#mi-", "");
      setScreen(initial || "dashboard", false);
    })();
  </script>
</section>
